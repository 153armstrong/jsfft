Sample data 
var data = new ComplexArray(128)
data.map(function(value, i, n) {
  value.real = (i > n/3 && i < 2*n/3) ? 1 : 0
})
Transformed:
fft_data = fft.FFT(data)
Simple low pass filter: 
fft_data.map(function(c_value, i, n) {
  if (i > n/5 && i < 4*n/5) {
    c_value.real = 0
    c_value.imag = 0
  }
})
Transformed back:
fft.InvFFT(fft_data)
... or all in one step 
fft.Filter(data, function(c_value, i, n) {
  if (i > n/5 && i < 4*n/5) {
    c_value.real = 0
    c_value.imag = 0
  }
})
